生成的 Swift 代码
class HomePageViewController: UIViewController {
    private let searchModuleView: UIView = {
        let view = UIView()
        view.backgroundColor = UIColor(hex: "#F2F3F5")
        view.layer.cornerRadius = 30
        return view
    }()

    private let inspirationStack: UIStackView = {
        let stack = UIStackView()
        stack.axis = .vertical  // flex-direction: column
        stack.backgroundColor = .white  // background: #FFFFFF
        return stack
    }()

    private let flashSaleView: UIView = {
        let view = UIView()
        view.backgroundColor = UIColor(hex: "#FFD700")
        view.layer.cornerRadius = 12
        return view
    }()
}
